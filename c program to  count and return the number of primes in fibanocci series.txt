//c program to  count and return the number of primes in fibanocci series
#include <stdio.h>

int isPrime(int n) 
{
    if (n <= 1)
       return 0;
    for (int i = 2; i * i <= n; i++)
    {
        if (n % i == 0) 
           return 0;
    }
    return 1;
}

int countPrimeFibonacci(int n) 
{
    int a = 0, b = 1, count = 0, temp;
    while (a <= n) 
    {
        if (isPrime(a)) 
          count++;
        temp = a + b;
        a = b;
        b = temp;
    }
    return count;
}

int main() 
{
    int n;
    printf("Enter a number: ");
    scanf("%d", &n);
    printf("Prime Fibonacci terms â‰¤ %d = %d\n", n, countPrimeFibonacci(n));
    return 0;
}

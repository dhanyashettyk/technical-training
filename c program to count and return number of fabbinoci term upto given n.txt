//c program to count and return number of fabbinoci term upto given n
#include <stdio.h>

int countFibonacciUptoN(int n) 
{
    int a = 1, b = 1, count = 0, temp; 
    while (a <= n)
    {                   
        temp = a + b;
        a = b;         
        b = temp;          
    }
    return count;
}

int main()
{
    int n;
    printf("Enter a number: ");
    scanf("%d", &n);
    printf("Fibonacci terms â‰¤ %d = %d\n", n, countFibonacciUptoN(n));
    return 0;
}
